cmake_minimum_required(VERSION 3.16)

project(AvionicsHealthMonitor VERSION 1.0 LANGUAGES CXX)


set(CMAKE_EXPORT_COMPILE_COMMANDS ON)


# -----------------------------
# C++ STANDARD
# -----------------------------
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)


# -----------------------------
# MAIN LIBRARY (CORE SYSTEM)
# -----------------------------
add_library(avionics_core
    src/SensorSimulator.cpp
    src/HealthMonitor.cpp
    src/Logger.cpp
    src/AlertManager.cpp
    src/TelemetryProcessor.cpp
)

target_include_directories(avionics_core
    PUBLIC
        ${CMAKE_SOURCE_DIR}/include
)

target_compile_options(avionics_core PRIVATE
    -Wall -Wextra -Wpedantic
)

# -----------------------------
# EXECUTABLE
# -----------------------------
add_executable(avionics_monitor
    src/main.cpp
)

target_link_libraries(avionics_monitor
    PRIVATE avionics_core
)

# -----------------------------
# OPTIONAL: clean logs folder warning
# -----------------------------
file(MAKE_DIRECTORY ${CMAKE_SOURCE_DIR}/logs)

include(FetchContent)

FetchContent_Declare(
    googletest
    URL https://github.com/google/googletest/archive/refs/tags/v1.17.0.zip
)

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)

FetchContent_MakeAvailable(googletest)

enable_testing()

add_subdirectory(tests)
